class parent;
	int a, b;
	virtual task display_c();
		$display("Parent");
	endtask
endclass

class child extends parent;
	int a;
	task display_c();
		$display("Child");
	endtask
endclass

program ldy;
initial begin
	parent p;
	child c;
	c = new();
	p = c;
	c.display_c();   //result: Child        Child
	p.display_c();   //result:Parent       Child
end

endprogram
